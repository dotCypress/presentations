Invocation


db.articles.aggregate(
  { $project : {
     author : 1,
     tags : 1,
  }},
  { $unwind : "$tags" },
  { $group : {
     _id : { tags : "$tags" },
     authors : { $addToSet : "$author" }
  }}
);

$project


db.article.aggregate({
$project: {
	_id: 0 ,
	title: 1 ,
	author: 1,
	page_views: "$pageViews",
	bar: "$other.foo",
	views: { $add:["$pageViews", 10]}
}});
